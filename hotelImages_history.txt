+/**
+ * Get main image for a hotel. Falls back to a themed image if ID not in map.
+ */
+export const getHotelMainImage = (hotelId: number): string => {
+  if (HOTEL_IMAGES[hotelId]) return HOTEL_IMAGES[hotelId].main;
+  // Fallback based on ID mod so we always get different images
+  const fallbacks = [
+    u("photo-1542314831-068cd1dbfeeb"),
+    u("photo-1566073771259-6a8506099945"),
+    u("photo-1551882547-ff40c63fe2e2"),
+    u("photo-1529290130-4ca3753253ae"),
+    u("photo-1611892440504-42a792e24d32"),
+    u("photo-1613395877344-13d4a8e0d49e"),
+    u("photo-1540541338287-41700207dee6"),
+    u("photo-1564501049412-61c2a3083791"),
+  ];
+  return fallbacks[hotelId % fallbacks.length];
+};
+
+/**
+ * Get gallery images for a hotel's detail page.
+ */
+export const getHotelGallery = (hotelId: number): string[] => {
+  if (HOTEL_IMAGES[hotelId]) return HOTEL_IMAGES[hotelId].gallery;
+  return [getHotelMainImage(hotelId)];
+};
